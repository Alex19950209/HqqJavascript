        function Stamp(timestamp){
            if(!timestamp) return 'timestamp is non-existent';
            timestamp = typeof timestamp == 'number' ? timestamp.toString():timestamp;
            if(timestamp.length< 10 || timestamp.length > 13) return 'timestamp error';
            timestamp = timestamp.length == 10 ? parseInt(timestamp) * 1000 : parseInt(timestamp);
            var stamptime,nowtime,year,month,day,hour,min,sec;
            var double = function (obj){
                return obj = obj < 10 ? '0' + obj : obj;
            }
//            if(!Date.now()){
//                nowtime = new Date().getTime();
//            }else{
//                nowtime = Date.now();
//            }
            nowtime = new Date();            
            stamptime = new Date(timestamp);
            year =  stamptime.getFullYear();
            month = stamptime.getMonth() + 1;
            day = stamptime.getDate();
            hour = stamptime.getHours();
            hour = stamptime.getHours();
            min = stamptime.getMinutes();
            sec = stamptime.getSeconds();
            if(nowtime.getFullYear() !== year){
                return year+ '/' + month + '/' + day + ',' + double(hour) + ':' + double(min) + ':' + double(sec);
            }else{
                if(nowtime.getMonth() + 1 != month){
                    return month + '/' + day;
                }else{
                    if(nowtime.getDate() > day){
                        return month + '/' + day;
                    }else if(nowtime.getDate() == day){
                        return double(hour) + ':' + double(min) + ':' + double(sec);
                    }else{
                        return month + '/' + day + ',' + double(hour) + ':' + double(min) + ':' + double(sec);
                    }
                }
            }
            
        }
